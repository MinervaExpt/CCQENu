185 std::map<const T*,int> UniverseDecoder(const std::map< std::string, std::vector<T*> > univs)const{
186 std::map<const T*, int> decoder;
187 for(auto univ:univs){
188 std::string name = univ.first;
189 std::vector<T*> pointers = univ.second;
190 for (int i = 0; i < pointers.size(); i++){
191 decoder[pointers[i]] = i;
192 }
193 }
194 return decoder;
