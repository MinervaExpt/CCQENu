33double MultiScaleFactors::DoEval(const double* parameters) const{
34 //For now just copying in the chi^2 function more or less directly from Andrew
35 double chi2 = 0.0;
36 if (!fDoFit) return chi2;
37 double fitSum;
38#ifdef DEBUG
39 std::cout << "parameters ";
40 for (int i = 0; i < fNdim; i++){
41 std::cout << parameters[i] << " ";
42 }
43 std::cout << std::endl;
44#endif
45 for (auto const sample:fUnfitHists){ // loop over samples
46 std::string whichsample = sample.first;
47 if (!fInclude.at(whichsample)) continue; // skip some samples
48 for (int whichBin = fFirstBin; whichBin <= fLastBin; ++whichBin){
49 double fitSum = 0.0;
50 for (int whichFit = 0; whichFit < fNdim; whichFit++){
51 double temp = fUnfitHists.at(whichsample).at(whichFit)->GetBinContent(whichBin)*parameters[whichFit];
52 fitSum += temp;
53 }
54 double diff;
55 double dataErr=0.0;
56 double dataContent = fDataHist.at(whichsample)->GetBinContent(whichBin);
57 if (fType == kFastChi2){
58 dataErr = fDataHist.at(whichsample)->GetBinError(whichBin);
59 diff = fitSum-dataContent;
60 if (dataErr > 1e-10) chi2 += (diff*diff)/(dataErr*dataErr);
61 }
62 if (fType == kSlowChi2){ // use the MC as a better estimator
63 double MCval = fitSum>0?fitSum:-fitSum;
64 diff = fitSum-dataContent;
65 chi2 += (diff*diff)/MCval;
66 }
67 if (fType == kML){
68 diff = fitSum-dataContent;
69 chi2 -= 2.*TMath::Log(TMath::Poisson(dataContent,fitSum));
70 }
71#ifdef DEBUG
72 std::cout << whichsample << "Fit Sum: " << fitSum << ", Data: " << dataContent << ", Difference: " << diff << ", Error: " << dataErr << std::endl;
73#endif
74 }
75 }
76
77#ifdef DEBUG
78 std::cout << "About to return chi2 of: " << chi2 << std::endl;
79#endif
80 return chi2;
81}
