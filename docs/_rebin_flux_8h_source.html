24// DeWidth -- utility to remove binwidth correction from an MnvH1D
25void DeWidth(MnvH1D* h_mnv_with_binwidth, MnvH1D *&h_mnv_with_binwidth_removed){
26
27 //cout << "TRACE: remove binwidth correction from MnvH1D " << h_mnv_with_binwidth->GetName() << endl;
28
29 //strategy is to recale orig by bin width (undo bin width normalization) then combine bins and then rescale by binwidth again
30 MnvH1D* scaler = (MnvH1D*)h_mnv_with_binwidth->Clone("widths");
31 scaler->SetDirectory(0);
32 std::string name = h_mnv_with_binwidth->GetName();
33 name += std::string("_counts");
34
35 double flux = 0.0;
36 for(int i=0;i<scaler->GetNbinsX()+2;i++) {
37 scaler->SetBinContent(i,scaler->GetBinWidth(i));
38 flux += scaler->GetBinWidth(i)*h_mnv_with_binwidth->GetBinContent(i);
39 // std::cout << "bin " << i << " " << scaler->GetBinCenter(i) << " " << scaler->GetBinWidth(0) << " " << h_mnv_with_binwidth->GetBinContent(i)<< " " << flux << std::endl;
40 }
41 h_mnv_with_binwidth_removed = (MnvH1D*)h_mnv_with_binwidth->Clone(name.c_str());
42 h_mnv_with_binwidth_removed->SetDirectory(0);
43 h_mnv_with_binwidth_removed->MultiplySingle(h_mnv_with_binwidth,scaler, 1., 1.);//undid bin width normalization
44 //std::cout << "integral for mnv after removing binwidth correction " << h_mnv_with_binwidth_removed->GetName() << " is " << h_mnv_with_binwidth_removed->Integral() << std::endl;
45}

49 HistoType1D* RebinDeWidthedFluxHist(const HistoType1D* h_mnv_without_binwidth, const HistoType1D * h_mnv_rebinned_without_binwidth){
50 // this operates on a dewidthed as we want total counts.
51 //cout << "TRACE: Rebin an MnvH1D " << h_mnv_without_binwidth->GetName() << " into " << h_mnv_rebinned_without_binwidth->GetName() << endl;
52 std::vector<double>mnv_rebinned_without_binwidth_bin_edges;
53 for(int i=1;i<h_mnv_rebinned_without_binwidth->GetNbinsX()+2;i++){//need low edge of overflow (high edge of last bin)
54 mnv_rebinned_without_binwidth_bin_edges.push_back(h_mnv_rebinned_without_binwidth->GetBinLowEdge(i));
55 }
56 HistoType1D* not_rebinned = (HistoType1D*)h_mnv_without_binwidth->Clone(); // have to do this as otherwise the rebin occurs multiple times
57 HistoType1D* new_rebinned = (HistoType1D*)h_mnv_rebinned_without_binwidth->Clone();
58 new_rebinned->SetDirectory(0);
59 new_rebinned = (HistoType1D*)not_rebinned->Rebin(mnv_rebinned_without_binwidth_bin_edges.size()-1,"Rebinned",&mnv_rebinned_without_binwidth_bin_edges[0]);
60 new_rebinned->SetName("flux");
61 //new_rebinned->Print("ALL");
62 //cout << "TRACE: Rebin an MnvH1D " << h_mnv_without_binwidth->GetName() << " into " << new_rebinned->GetName() << endl;
63 return new_rebinned;
