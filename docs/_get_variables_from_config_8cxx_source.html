43std::map<std::string, CCQENu::VariableFromConfig*> GetVariablesFromConfig(const std::vector<std::string> vars, const std::vector<std::string> tags, NuConfig configraw) {
44 //=========================================
45 // Constructor wants: name, x-axis label,
46 // binning,
47 // CVUniverse reco and
48 // truth functions
49 //=========================================
50 std::map<std::string, CCQENu::VariableFromConfig*> allvariables; // set this internally to span the set of variables
51 std::map<std::string, CCQENu::VariableFromConfig*> variablesmap; // this is the set that actually gets returned
52
53
54
55// if (configfilename != "" ){
56// // NuConfig config;
57// // config.Read(configfilename);
58// NuConfig configraw;
59// configraw.Read(configfilename);
60 NuConfig config;
61 if(configraw.IsMember("1D")){
62 config = configraw.GetValue("1D");
63 } else config = configraw;
64 std::vector<std::string> keys = config.GetKeys();
65
66 for (auto key:keys){ // this is in the Variables json file so you can define them all
67 bool found = false;
68 for ( auto v:vars){ // this is from the master driver file so you can skip some
69 if ( v == key){
70 found = true;
71 allvariables[key] = new CCQENu::VariableFromConfig( config.GetValue(key));
72 std::cout << "GetVariables: set up variable " << allvariables[key]->GetName() << std::endl;
73 }
74 }
75 if(!found){
76 std::cout << "GetVariables: variable " << key << " configured but not requested" << std::endl;
77 }
78 }
79
80
81
82 // ok - have made all of the possible variables, but now let's choose based on the top level config.
83
84 std::map<std::string,bool> amIused;
85
86 for (auto variable:allvariables){
87 amIused[variable.first]=false;
88 }
89
90 for (auto var:vars){
91 bool found = false;
92 for (auto variable:allvariables){
93 // std::cout << var << " check " << variable.first<< std::endl;
94 if(var == variable.first){
95 std::cout << "GetVariables: study 1D variable " << var << std::endl;
96 // this is the point where you add the tags. Saves space this way.
97 variable.second->AddTags(tags); //
98 variablesmap[var] = variable.second;
99 found = true;
100 amIused[variable.first] = true;
101 break;
102 }
103 }
104 if (!found){
105 std::cout << "GetVariablesFromConfig: Warning - have requested an unimplemented variable in GetVariablesFromConfig " << var << std::endl;
106 assert(0);
107 }
108 }
109 // clean up unused variables
110 for (auto variable:allvariables){
111 if (amIused[variable.first]) continue;
112 std::cout << "GetVariables: remove unused variable " << variable.first << std::endl;
113 }
114 return variablesmap;
115}
