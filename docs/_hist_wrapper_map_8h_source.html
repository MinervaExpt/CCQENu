112 std::map<const T*,int> UniverseDecoder(const std::map< std::string, std::vector<T*> > univs)const{
113 std::map<const T*, int> decoder;
114 for(auto univ:univs){
115 std::string name = univ.first;
116 std::vector<T*> pointers = univ.second;
117 for (int i = 0; i < pointers.size(); i++){
118 decoder[pointers[i]] = i;
119 }
120 }
121 return decoder;
122 }
