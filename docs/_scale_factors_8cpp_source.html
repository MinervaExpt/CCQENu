12 ScaleFactors::ScaleFactors(const std::vector<TH1D*> fitHists, const std::vector<TH1D*> unfitHists, const TH1D* dataHist,
13 const int firstBin, const int lastBin):IBaseFunctionMultiDimTempl<double>(), fFitHists(fitHists),
14 fUnfitHists(unfitHists), fDataHist(dataHist), fFirstBin(firstBin), fLastBin(lastBin), fDoFit(true)
15 {
16 if (fFitHists.size() == 0){
17 std::cout << "Function has no histos to fit... Setting the fit condition to false." << std::endl;
18 fDoFit = false;
19 }
20
21 int nBins = 0;
22 if (fDoFit) nBins = fFitHists.at(0)->GetNbinsX();
23
24 for(unsigned int iFit; iFit < fFitHists.size(); ++iFit){
25 if (!fDoFit) break;
26 if (fFitHists.at(iFit)->GetNbinsX() != nBins){
27 std::cout << "No. of bins not consistent... Setting the fit condition to false." << std::endl;
28 fDoFit = false;
29 }
30 }
31
32 for(unsigned int iFit; iFit < fUnfitHists.size(); ++iFit){
33 if (!fDoFit) break;
34 if (fUnfitHists.at(iFit)->GetNbinsX() != nBins){
35 std::cout << "No. of bins not consistent... Setting the fit condition to false." << std::endl;
36 fDoFit = false;
37 }
38 }
39
40 if (fDataHist->GetNbinsX() != nBins){
41 std::cout << "No. of bins not consistent... Setting the fit condition to false." << std::endl;
42 fDoFit = false;
43 }
44
45 //Avoid fitting overflow
46 if (fDoFit && fLastBin < 0 || fLastBin > nBins) fLastBin=nBins;
47 }

50 if (!fDoFit) return 0;
51 else return fFitHists.size();

54 double ScaleFactors::DoEval(const double* parameters) const{
55 //For now just copying in the chi^2 function more or less directly from Andrew
56 double chi2 = 0.0;
57 if (!fDoFit) return chi2;
58
59 double fitSum = 0.0;
60 /*std::cout << "parameters ";
61 for (int i = 0; i < fFitHists.size(); i++){
62 std::cout << parameters[i] << " ";
63 }
64 std::cout << std::endl;*/
65 for (int whichBin = fFirstBin; whichBin <= fLastBin; ++whichBin){
66 double fitSum = 0.0;
67 for(unsigned int whichFit=0; whichFit < fFitHists.size(); ++whichFit){
68
69 double temp = fUnfitHists.at(whichFit)->GetBinContent(whichBin)*parameters[whichFit];
70 fitSum += temp;
71 }
72
73 double dataContent = fDataHist->GetBinContent(whichBin);
74 double dataErr = fDataHist->GetBinError(whichBin);
75 double diff = fitSum-dataContent;
76
77 //std::cout << "Fit Sum: " << fitSum << ", Data: " << dataContent << ", Difference: " << diff << ", Error: " << dataErr << std::endl;
78 //std::cout << "How the chi2 should change: " << (diff*diff)/(dataErr*dataErr) << std::endl;
79 if (dataErr > 1e-10) chi2 += (diff*diff)/(dataErr*dataErr);
80 //std::cout << "Updated chi2: " << chi2 << std::endl;
81 }
82
83
84 //std::cout << "About to return chi2 of: " << chi2 << std::endl;
85
86 return chi2;
87 }
